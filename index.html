# @title 🔗 연결리스트 끝말잇기 센터 { run: "auto" } import IPython from IPython.display import HTML, display, clear_output # 1. 연결리스트 노드 정의 class WordNode: def __init__(self, word): self.word = word self.next = None # 2. 게임 관리 클래스 class WordChainGame: def __init__(self): self.head = WordNode("기차") # 시작 단어 self.history = ["기차"] self.msg = "첫 단어는 '기차'입니다. 끝말을 이어보세요!" self.msg_color = "#2ecc71" def add_word(self, new_word): if not new_word or len(new_word) < 2: self.msg = "⚠️ 두 글자 이상 입력해주세요!" self.msg_color = "#e67e22" return if new_word in self.history: self.msg = f"🚫 '{new_word}'는 이미 사용한 단어입니다!" self.msg_color = "#e74c3c" return # 마지막 노드 찾기 (Traversal) curr = self.head while curr.next: curr = curr.next # 규칙 검사 (연결리스트의 데이터 검증) if curr.word[-1] == new_word[0]: curr.next = WordNode(new_word) # 새 노드 연결 (Link) self.history.append(new_word) self.msg = f"✅ '{new_word}' 연결 성공!" self.msg_color = "#2ecc71" else: self.msg = f"❌ 틀렸습니다! '{curr.word[-1]}'로 시작해야 합니다." self.msg_color = "#e74c3c" def render(self): clear_output() # 기차(연결리스트) 시각화 HTML 생성 train_html = "" curr = self.head while curr: train_html += f'
{curr.word}
' if curr.next: train_html += '
➡
' curr = curr.next # 전체 레이아웃 display(HTML(f"""

🧩 끝말잇기 연결리스트

{self.msg}

{train_html}
""")) # 3. 게임 인스턴스 생성 및 파이썬-자바스크립트 연결 game = WordChainGame() def add_word_js(new_word): game.add_word(new_word) game.render() from google.colab import output output.register_callback('notebook.add_word_js', add_word_js) # 초기 화면 렌더링 game.render()
